
#Loading multiple pictures from multiple folders 

import os
import numpy as np
from PIL import Image
from sklearn.preprocessing import LabelEncoder

# Define the path to the folder containing the subfolders
data_dir = os.path.join(os.path.expanduser('~'), 'Downloads')

# Define the image dimensions
img_width, img_height = 224, 224

# Create an empty list to store the image data and labels
X = []
y = []

# Loop through the subfolders and load the images
for folder in ['Transaction Input', 'Domestic Payment', 'SEPA']:
    folder_path = os.path.join(data_dir, folder)
    if os.path.isdir(folder_path):
        for img_file in os.listdir(folder_path):
            img_path = os.path.join(folder_path, img_file)
            img = Image.open(img_path)
            img = img.resize((img_width, img_height))
            img_data = np.array(img) / 255.0  # Normalize pixel values
            X.append(img_data)
            y.append(folder)

# Convert the data to NumPy arrays
X = np.array(X)
y = np.array(y)

# Encode the labels
label_encoder = LabelEncoder()
y = label_encoder.fit_transform(y)

print(f"Number of images: {len(X)}")
print(f"Number of classes: {len(np.unique(y))}")
